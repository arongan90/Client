1.테이블태그.html

<!doctype html>
<html>
 <head>
  <meta charset="UTF-8">
  <title>1.테이블태그.html</title>
 </head>
 <body>
 <center>
 <!-- 
  표를 작성 테두리->border=n(픽셀수),width,height(표의 크기)
                                                픽셀수/상대적인 단위 %
												70% ->양쪽여백 15% 균등
					   cellspacing->쉘과 쉘사이의 거리
					   cellpadding->쉘과 문자열사이의 거리
                       열(세로) valign=top,middle,bottom
 -->
  <table width="70%"  height="70%" border="1"
            cellspacing="0" cellpadding="0">
		<tr>
			<!-- <td align="center"><b>월</b></td>
			<td align="center"><b>수</b></td>
			<td align="center"><b>금</b></td> -->
			<th>월</th>
			<th>수</th>
			<th>금</th>
		</tr><!-- 행(가로) align->left,center,right -->
		<tr>
           <td align="left">왼쪽</td>
		   <td align="center">가운데</td>
		   <td align="right">오른쪽</td>
		</tr>
  </table>
  </center>
 </body>
</html>
